import pandas as pd
import matplotlib.pyplot as plt
import mpld3

# Create dataset
data = {
    "employee_id": ["EMP001","EMP002","EMP003","EMP004","EMP005"],
    "department": ["Operations","IT","Marketing","IT","R&D"],
    "region": ["Europe","Middle East","Europe","Europe","Middle East"],
    "performance_score": [61.49,77.67,88.34,85.38,88.62],
    "years_experience": [13,3,1,6,1],
    "satisfaction_rating": [3.9,3.7,3.1,4.1,4.3]
}
df = pd.DataFrame(data)

df.to_csv("employees.csv", index=False)

# Frequency count for Sales
sales_count = (df["department"] == "Sales").sum()
print("Frequency count for Sales department:", sales_count)

# Department distribution histogram
dept_counts = df["department"].value_counts().sort_index()
fig, ax = plt.subplots(figsize=(8,5))
ax.bar(dept_counts.index, dept_counts.values, color="skyblue", edgecolor="black")
ax.set_title("Department Distribution")
ax.set_xlabel("Department")
ax.set_ylabel("Count")
plt.xticks(rotation=45, ha="right")
